export const components = {
  pre: ({ children, ...props }) => {
    // 檢查是否為代碼區塊
    if (children?.props?.className && typeof children.props.children === 'string') {
      const language = children.props.className.replace('language-', '');
      const code = children.props.children;
      
      return (
        <CodeBlock language={language} showLineNumbers={true}>
          {code}
        </CodeBlock>
      );
    }
    
    // 如果children是字符串，直接使用
    if (typeof children === 'string') {
      return (
        <CodeBlock language="text" showLineNumbers={false}>
          {children}
        </CodeBlock>
      );
    }
    
    // 普通的 pre 標籤
    return <pre {...props}>{children}</pre>;
  },
  
  CodeBlock,
  AdvancedCodeBlock,
};

# 代碼高亮示範

## 基本語法高亮

使用標準的 Markdown 代碼區塊語法，現在會自動應用語法高亮：

```javascript
function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

console.log(fibonacci(10)); // 55
```

```python
def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    return quick_sort(left) + middle + quick_sort(right)
```

## 支援的主題

我們的進階代碼區塊支援以下主題：

- `vsDark` (預設) - VS Code 深色主題
- `vsLight` - VS Code 淺色主題  
- `github` - GitHub 風格
- `dracula` - Dracula 主題
- `oceanicNext` - Oceanic Next 主題
- `nightOwl` - Night Owl 主題

## 功能特色

✨ **語法高亮** - 支援多種程式語言  
📋 **一鍵複製** - 方便複製代碼  
📝 **行號顯示** - 可選擇顯示行號  
🎨 **多種主題** - 支援多種配色主題  
📁 **檔案名稱** - 可顯示檔案名稱和語言標籤  
🔍 **行高亮** - 可高亮特定行數  
🖥️ **終端機風格** - 仿 macOS 終端機的設計

## 使用方法

### 標準 Markdown 語法

\`\`\`javascript
// 你的代碼
\`\`\`

### 使用 CodeBlock 組件

\`\`\`jsx
<CodeBlock 
  language="javascript" 
  filename="app.js"
  showLineNumbers={true}
>
  // 你的代碼
</CodeBlock>
\`\`\`

### 使用 AdvancedCodeBlock 組件

\`\`\`jsx
<AdvancedCodeBlock 
  language="javascript" 
  title="我的應用程式"
  theme="dracula"
  showLineNumbers={true}
  highlightLines={[2, 5, 8]}
>
  // 你的代碼
</AdvancedCodeBlock>
\`\`\`
